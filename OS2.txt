#include<stdio.h> 
#include<stdlib.h> 
#include<unistd.h> 
#include<sys/types.h> 
#include<string.h> 
#include<sys/wait.h> 
int main() 
{  int fd1[2];  
    int fd2[2];  
    char str[100]; 
    pid_t p; 
  
 if (pipe(fd1)==-1) 
    { 
        fprintf(stderr, "Pipe Failed" ); 
        return 1; 
    } 
    if (pipe(fd2)==-1) 
    { 
        fprintf(stderr, "Pipe Failed" ); 
        return 1; 
    } 

    scanf("%s",str); 
    p = fork(); 
  
    if (p < 0) 
    { 
        fprintf(stderr, "fork Failed" ); 
        return 1; 
    } 
  
    else if (p > 0) 
    { 
        char str1[100]; 
  
        close(fd1[0]);  
        write(fd1[1], str, strlen(str)+1); 
        close(fd1[1]); 
        wait(NULL); 
  
        close(fd2[1]); 
        read(fd2[0], str1, 100); 
        printf("Concatenated string %s\n", str1); 
        close(fd2[0]); 
    } 

    else
    { 
        close(fd1[1]); 
        char str1[100]; 
        read(fd1[0], str1, 100); 
int c = 0;
   char ch;
    while (str1[c] != '\0') {
      ch = str1[c];
      if (ch >= 'A' && ch <= 'Z')
         str1[c] = str1[c] + 32;
      else if (ch >= 'a' && ch <= 'z')
         str1[c] = str1[c] - 32;  
      c++;  
   }
        close(fd1[0]); 
        close(fd2[0]); 
        write(fd2[1], str1, strlen(str1)+1); 
        close(fd2[1]); 
  
        exit(0); 
    } 
}
